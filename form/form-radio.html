# 单选

TIP

2.1.0+

# 普通用法

<avue-form :option="option" v-model="form"></avue-form>
<script>
export default {
    data() {
      return {
        form: {
          radio:0
        },
        option: {
          column: [{
              label: '单选',
              prop: 'radio',
              type: 'radio',
              span:24,
              dicData:[{
                label:'男',
                value:0
              },{
                label:'女',
                value:1
              },{
                label:'未知',
                value:''
              }]
            }
          ]
        }
      }
    }
}
</script>
